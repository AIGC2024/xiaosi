[root@localhost ~]# man sort | cat
...
SYNOPSIS
       sort [OPTION]... [FILE]...
       sort [OPTION]... --files0-from=F
...

OPTION
--version
    # 输出版本并退出
-n, --numeric-sort
    # 根据值比较
-r, --reverse
    # 反转比较结果，从升序改为倒序
-c, --check, --check=diagnose-first
    # 查看是否排序
-u, --unique
    # 去除重复，如果与 -c 选择同时使用，则先去重，再排序
-k, --key=KEYDEF
    # 根据 key 排序
    # KEYDEF为F[.C][OPTS][,F[.C][OPTS]] 起止位置，
    # 其中F为字段号，C为字符在字段中的位置； 均为原点 1，停止位置默认为行尾。 
    # 如果 -t 和 -b 均未生效，则字段中的字符从前面的空白开始计算。 
    # OPTS 是一个或多个单字母排序选项 [bdfgiMhnRrV]，
    # 它覆盖该键的全局排序选项。 如果没有给出键，则使用整行作为键。
-t, --field-separator=SEP
    # 指定字段分隔符，默认为一个或多个空白符（非空白到空白）

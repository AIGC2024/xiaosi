# autcomd 是自动事件
# BufReadPost读取文件之后，BufNewFile创建文件时
# ~/nginxDocker 是nginx配置文件目录
# source 指定 nginx.vim 位置

" 定义nginx配置文件
au BufRead,BufNewFile *.nginx exec ":source ~/vimrc/nginx.vim"
au BufRead,BufNewFile *.nginx.conf exec ":source ~/vimrc/nginx.vim"
au BufRead,BufNewFile nginx.conf exec ":source ~/vimrc/nginx.vim"

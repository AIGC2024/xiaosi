[root@centos7 nginx-1.18.0]# ln  /usr/local/nginx/sbin/nginx /usr/bin/nginx

[root@centos7 nginx-1.18.0]# cat << EOF > /usr/lib/systemd/system/nginx.service
[Unit]
# 描述名称
Description=The nginx HTTP and reverse proxy server
# 服务类别
After=network.target remote-fs.target nss-lookup.target

[Service]
# 运行方式，forking 表示后台运行
Type=forking
# pid 文件
PIDFile=/usr/local/nginx/logs/nginx.pid
# 启动 nginx 之前执行的命令
ExecStartPre=/usr/bin/rm -f /usr/local/nginx/logs/nginx.pid
ExecStartPre=/usr/local/nginx/sbin/nginx -t
# 启动 nginx 执行的命令
ExecStart=/usr/local/nginx/sbin/nginx
# 重新加载 nginx 配置文件的命令
ExecReload=/usr/local/nginx/sbin/nginx -s reload
# 关闭 nginx 执行的命令
ExecStop=/usr/local/nginx/sbin/nginx -s stop
KillSignal=SIGQUIT
TimeoutStopSec=5
KillMode=process
# 分配独立的临时空间
PrivateTmp=true
Restart=on-failure
RestartSec=42s

[Install]
WantedBy=multi-user.target
EOF

# 复制镜像中的配置文件
[root@xiaosi bash]# cp -r  /var/lib/docker/overlay2/81e5129d2be22c4bf449c9c5ff7b82b84a545584dba08f515fd95c938e21910b/diff/etc/nginx/* /root/nginxDocker/nginx/conf/

# 创建 nginx docker 脚本 
[root@xiaosi bash]# cat nginxDocker.sh 
docker network create --subnet 172.11.11.0/24 net-nginx 2>/dev/null
# /etc/nginx 是配置文件目录
docker container run --name nginx \
	-v /etc/localtime:/etc/localtime \
	-v /root/nginxDocker/nginx/conf/:/etc/nginx \
	-v /root/nginxDocker/nginx/web/:/nginx/web \
	-v /root/nginxDocker/nginx/log/:/nginx/log \
	--network net-nginx \
	--ip 172.11.11.2 \
	-d nginx

# 创建 nginx docker
[root@xiaosi ~]# bash bash/nginxDocker.sh 
f5fa088141fa73f9b9b561b4412d2d792192199f6f9a489a3e45ff671852de81

# 由于容器环境有些缺省值不一样，最好在配置文件中明确指定
# 如 root 在容器中缺省值为 /etc/nginx/html
[root@xiaosi ~]# cat /root/nginxDocker/nginx/conf/conf.d/1001-linuxInfo.conf 

server {
    listen 172.11.11.2:1001;
	location = /linuxInfo {
	 return 200 "$document_root \n";
	}
}

# 重新加载配置文件
[root@xiaosi ~]# docker container exec -it nginx nginx -s reload 
2021/09/11 16:03:28 [notice] 50#50: signal process started

# 输出
[root@xiaosi ~]# curl 172.11.11.2:1001/linuxInfo
/etc/nginx/html 

# 可以创建别名快速使用命令
[root@xiaosi ~]# alias nginx='docker container exec -it nginx nginx'

[root@xiaosi /]# nginx -v

# 开机生效
[root@xiaosi /]# echo "alias nginx='docker container exec -it nginx nginx'" >> ~/.bashrc

-t			# 检查配置文件

-g			# 修改配置文件中的指令
-s			# 执行操作以下
	stop: 立即停止服务
	quit: 优雅停止服务
	reload: 重新加载配置文件
	reopen: 重新开始记录日志文件

热部署

步骤：

  1. 备份当前二进制nginx文件
  2. 把编译之后的二进制文件（objs/nginx）替换现有的nginx二进制文件 （nginx命令文件）
  3. 向nginx master 进程发送 USR2信号，如kill -USR2 1192。 系统会根据新的nginx二进制文件启动master处理新的连接， 旧的master进程还在处理已经建立的连接
  4. 向旧的master进程发送信号使其优雅的关闭连接（kill -WINCH 1192）

流程：

  1. 将旧nginx文件换成新的nginx文件（注意备份）
  2. 向master进程发送USR2信号
  3. master进程再pid文件添加后缀.oldbin
  4. master进程用新的nginx文件启动新的master进程
  5. 向旧的master进程发送QUIT信号，关闭master老进程

回滚：

  1. 向旧master进程发送HUP信号
  2. 向信master进程发送QUIT信号

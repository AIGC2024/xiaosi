使用 rmdir 命令实现

rmdir 命令很简单，判断目录是是否有内容，无内容则删除，只能递归空目录

1
2
3
4

# -p 选项只能递归空目录
[root@localhost ~]# rmdir -p /usr/local/nginx/note/
# 要借助通配符才能实现递归删除空目录
[root@localhost ~]# rmdir -p /usr/local/nginx/note/*/*/*/*/*

使用 find 命令实现

1
2
3
4

# -type 指定为目录类型
# -empty 只对空目录进行限制
# -delete 执行删除操作，包括子目录在内的所有空目录
[root@localhost ~]# find /usr/local/nginx/note/ -type d -empty -delete
